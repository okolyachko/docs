   self.button.onEvent = ^(NSString *payload) {
     NSData* eventData = [payload dataUsingEncoding:NSUTF8StringEncoding];
     NSError* error = nil;
     NSDictionary* eventDict = [NSJSONSerialization JSONObjectWithData:eventData options:0 error:&error];
     if (error != nil) {
       return;
     }
     NSLog(@"%@", eventDict);
   };
