   self.button.onEvent = { event in
     guard let eventData = event?.data(using: .utf8, allowLossyConversion: false),
       let eventDict = try? JSONSerialization.jsonObject(with: eventData, options: .mutableContainers)
     else {
       return
     }
     print(eventDict)
   }
